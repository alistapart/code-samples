            agent_isAOL= (agent.indexOf("AOL") >= 0); 

            if (agent_isIE || agent_isNS) {
                if (agent_isIE && version.indexOf("MSN") == -1) {
                    
                    agent_Major = version.substr(version.indexOf("MSIE") + 5,version.indexOf("."));
                    agent_Major = parseInt(agent_Major); 
                }
                if (agent_isNS) 
                    agent_Major = parseInt(version.substr(0,2));

                if (agent_Major >= 4) bScreen = true;   
                agent_isWin = (agent.indexOf("Win") >= 0);
                agent_isMac = (agent.indexOf("Mac") >= 0);
            }
            agent_isGoldenVersion = ( agent_isWin && agent_isIE && agent_Major>=4);
        }
