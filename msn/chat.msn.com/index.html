function GetCookie(name) 
{
	var result = null;
	var myCookie = " " + document.cookie + ";";
	var searchName = " " + name + "=";
	var startOfCookie = myCookie.indexOf(searchName);
	var endOfCookie;
	if (startOfCookie != -1) 
	{
		startOfCookie += searchName.length; 
		endOfCookie = myCookie.indexOf(";", startOfCookie);
		result = unescape(myCookie.substring(startOfCookie, endOfCookie));
	}
	return result;
}

function SetCookie(name, value) 
{
	if (name)
	{	
		var never = new Date();

		if (value)
		{
			never.setDate (31); 
			never.setMonth (12); 
			never.setYear (2037);
		}
		else 
			never.setTime(never.getTime() - 1);

		var expString = "; expires=" + never.toGMTString();
		document.cookie = name + "=" + escape(value) + expString + "; path=/";
	}
}

function IsCookieEnabled() 
{
	SetCookie("testKey", "1");

	if (GetCookie("testKey") == "1")
		return 1;

	return 0;
}

function GetChatList(maxrooms)
{
	var i = 0;

	if (maxrooms == 0)
		maxrooms = 6;

	if (cookie)
	{
		var len = cookie.length;
		var pos = 1;
		//while (pos>0 && i<len && cookie.length>1)
		while (pos>0 && i<maxrooms && cookie.length>1)
		{
			//createdBy[i] = cookie.substring(0,1);

			cookie = cookie.substring(1,cookie.length);

			pos = cookie.indexOf("\"");
			if (pos > 0)
				s = cookie.substring(0,pos);
			else
				s = cookie.substring(pos+1,cookie.length);

			//get cat name and hex name
			catPos = s.indexOf("|");
			if (catPos != -1)
			{
				room[i] = s.substring(0, catPos);
				hexPos = s.indexOf("|", catPos+1);
				if (hexPos == -1)
					hexPos = cookie.length;

			else
			{
				room[i] = s;
			}

			i++;
			cookie = cookie.substring(pos+1,cookie.length);
		};
	}

	return i;
